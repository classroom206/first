	AREA SMALLEST, CODE, READONLY
	EXPORT __main
		
ARRAY1 DCD 5
	DCD 4
	DCD 10
	DCD 1
	DCD 8
		
__main
	LDR R0,=ARRAY1
	LDR R1,=5
	LDR R3,[R0],#4
	SUBS R1,R1,#1
LOOP
	LDR R4,[R0],#4
	CMP R3,R4
	BLS SMALLER
	MOV R3,R4
SMALLER 
	SUBS R1,R1,#1
	BNE LOOP
	LDR R5,=SMALLENO
	STR R3,[R5]
	NOP
	NOP
	
	AREA DATA1, DATA, READWRITE
SMALLENO DCD 0x0
	END
