	AREA ADDITION64, CODE, READONLY
	EXPORT __main

VAL1 DCD 0x11111111
VAL2 DCD 0x22222222
VAL3 DCD 0x33333333
VAL4 DCD 0x44444444
	
__main
	LDR R0,VAL1
	LDR R1,VAL2
	LDR R2,VAL3
	LDR R3,VAL4
	ADDS R4,R0,R2
	ADC R5,R1,R3
	LDR R6,=LOWER
	STR R4,[R6]
	LDR R7,=HIGHER
	STR R5,[R7]
	
STOP B  STOP

	AREA DATA1, DATA, READWRITE

LOWER DCD 0x0
HIGHER DCD 0x0

	END
